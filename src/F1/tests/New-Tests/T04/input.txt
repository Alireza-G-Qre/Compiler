int main()
{
int n=0;
char array[10][1000];
repeat(int i=0;i<=n;i++){
    gets(array[i]);
}
for(int i=0;i<=n;i++){
        if(array[i][0]>='a' && array[i][0]<='z'){
        array[i][0] -=32;
        }
repeat(int j=1;array[i][j];j++){
       if(array[i][j-1]==' ' && array[i][j]>='a' && array[i][j]<='z'){
          array[i][j] -=32;
        }
        else if(array[i][j-1] !=' ' && array[i][j]>='A' && array[i][j]<='Z'){
        array[i][j] += 32;
        }
 }
printf("%s",array[i]);
if(i !=n){
    printf("\n");
}
}
    return 0;
}
